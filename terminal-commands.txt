# 1. Install Helm (if not already installed)
winget install Helm.Helm

# 2. Add and Update Traefik Repository
helm repo add traefik https://traefik.github.io/charts
helm repo update

# 3. Install Traefik to port 8081 (to avoid Windows IIS conflict)
helm install traefik traefik/traefik --create-namespace   --namespace traefik  --set ports.web.exposedPort=8081

# 4. Deploy your V1 and V2 Applications
# (Ensure you have your apps.yaml file in the current folder)
kubectl apply -f apps.yaml

# 5. Apply the 80/20 Traffic Split Rules
# (Ensure you have your split-test.yaml file in the current folder)
kubectl apply -f split-test.yaml

# 6. Apply the Dashboard Access Rule
# (Ensure you have your traefik-dashboard.yaml file in the current folder)
kubectl apply -f traefik-dashboard.yaml

# 7. Test the 80/20 split (Run 10 times)
1..10 | ForEach-Object { curl.exe -s -H "Host: split.localhost" http://localhost:8081 }


helm upgrade traefik traefik/traefik   --namespace traefik  --reuse-values  --set "additionalArguments={--api.dashboard=true,--api.insecure=true}"
helm upgrade traefik traefik/traefik   --namespace traefik   --reuse-values   --set "additionalArguments={--metrics.prometheus=true,--api.dashboard=true,--api.insecure=true}"
kubectl port-forward $(kubectl get pods -n traefik -l app.kubernetes.io/name=traefik -o name) 9000:8080 -n traefik
